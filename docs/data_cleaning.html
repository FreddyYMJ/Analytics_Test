# 2. FUNCIÓN PARA CARGAR TABLAS
def cargar_tabla(nombre_tabla, limite=None):
    """Carga una tabla desde SQL Server"""
    try:
        with engine.connect() as conn:
            if limite:
                query = text(f"SELECT TOP {limite} * FROM [dbo].[{nombre_tabla}]")
            else:
                query = text(f"SELECT * FROM [dbo].[{nombre_tabla}]")
            df = pd.read_sql(query, conn)
            print(f" {nombre_tabla}: {len(df):,} filas, {len(df.columns)} columnas")
            return df
    except Exception as e:
        print(f" Error en {nombre_tabla}: {e}")
        return None

def aj_cantidad(row):
    if row['ImRecCod'] == "BONOS":
        return 0
    if row['XmTDcCod'] in ["NDEB", "NDI", "NCREDV", "NCREDB", "NCREDE", "SFCLI", "FEXT"]:
        return 0
    if row['VdVenCla'] in ["Financiera", "Adicional", "VentaE", "VentaS", "MedContinua"]:
        return 0
    if row['VdVenCla'] == "Devolución":
        return row['VdVenCanD'] * -1
    if row['VdVenCla'] == "Anulación":
        return row['VdVenCan'] * -1
    return row['VdVenCanD']


df_VdVen['VdVenCanAj'] = df_VdVen.apply(aj_cantidad, axis=1)

6 6. Enriquecer tabla de Producción y agrupar con tabla PdGuia

Se agrupa la producción por número de guía para evitar duplicados. Se agregan las dimensiones de cada producto y se calcula el volumen producido.

Code 
print("PRODUCCIÓN AGRUPADA POR GUÍA")
print("-"*60)

# 1. Agrupar producción por guía
print("\n 1. Agrupando producción por guía...")
df_produccion = df_IdInv.groupby('IdInvGuia').agg(
    IdInvCan_sum = ('IdInvCan', 'sum'),
    ImRecCod = ('ImRecCod', 'first'),
    transacciones = ('IdInvCan', 'count')
).reset_index()

# Eliminar guías nulas
df_produccion = df_produccion.dropna(subset=['IdInvGuia'])
print(f"    Guías únicas: {len(df_produccion):,}")

# 2. Unir con PdGuia para obtener fecha
print("\n 2. Uniendo con PdGuia para obtener fecha...")
df_produccion = df_produccion.merge(
    df_PdGuia[['PdGuiaNum', 'PdGuiaFch']],
    left_on='IdInvGuia',
    right_on='PdGuiaNum',
    how='left'
)
df_produccion['fecha'] = pd.to_datetime(df_produccion['PdGuiaFch'], errors='coerce')
df_produccion = df_produccion.dropna(subset=['fecha'])
print(f"    Con fecha válida: {len(df_produccion):,} filas")

# 3. Unir con ImRec para dimensiones
print("\n 3. Uniendo con ImRec para dimensiones...")
df_produccion = df_produccion.merge(
    df_ImRec[['ImRecCod', 'ImRecLrg', 'ImRecAnc', 'ImRecAlt', 'ImRecGrp01', 'ImRecGrp04', 'ImRecGrp06']],
    on='ImRecCod',
    how='left'
)
print(f"    Dimensiones agregadas")

# 4. Calcular Prod_m3
print("\n 4. Calculando Prod_m3...")
df_produccion['Prod_m3'] = (
    df_produccion['IdInvCan_sum'] * 
    df_produccion['ImRecLrg'].fillna(1) * 
    df_produccion['ImRecAnc'].fillna(1) * 
    df_produccion['ImRecAlt'].fillna(1)
) / 1000

print(f"\n    Prod_m3 calculado")
print(f"    Total: {df_produccion['Prod_m3'].sum():,.2f} m³")
print(f"    Prod_m3 negativo: {(df_produccion['Prod_m3'] < 0).sum():,}")

# 5. Calcular turno (opcional)
print("\n 5. Calculando turno de producción...")

def calcular_turno(fecha_guia):
    if pd.isna(fecha_guia):
        return None
    hora = fecha_guia.hour + fecha_guia.minute / 60 + fecha_guia.second / 3600
    if 7 <= hora < 19:
        return 'T1'
    else:
        return 'T2'

if 'PdGuiaFch' in df_produccion.columns:
    df_produccion['Turno'] = df_produccion['PdGuiaFch'].apply(calcular_turno)
    print(f"    Turno calculado")
    print(f"       Distribución:")
    print(df_produccion['Turno'].value_counts())

# 6. Vista previa
print("\n Vista previa (primeras 3 filas):")
columnas_ver = ['fecha', 'IdInvGuia', 'ImRecCod', 'IdInvCan_sum', 'Prod_m3', 'Turno']
columnas_existentes = [col for col in columnas_ver if col in df_produccion.columns]
print(df_produccion[columnas_existentes].head(3))
